def calculadora(*args, **kwargs): valor_imovel = float(Element('Ivalor_imovel').element.value) entrada_valor = float(Element('Ientrada_valor').element.value) taxa_juros = float(Element('Itaxa_juros').element.value) prazo = float(Element('Iprazo').element.value) if valor_imovel <= 0 and entrada_valor <= 0: resultado = "Digite valores maiores que zero" else: valor_financiado = valor_imovel - entrada_valor taxa_juros_mensal = taxa_juros / 100 / 12 parcela = (valor_financiado * taxa_juros_mensal) / (1 - (1 + taxa_juros_mensal) ** -prazo) Element("resultado").element.innerHTML = parcela
